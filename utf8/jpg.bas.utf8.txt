Public Function DecodeHEXtoMemBloc(ByVal A$, ok As Boolean, Optional forcode As Boolean = False) As Object
    Dim mem As New MemBlock, BLen As Long
    A$ = DecodeHEX(A$, ok)
    If ok And LenB(A$) > 0 Then
        BLen = LenB(A$)
        mem.Construct 1, BLen, , forcode
        CopyBytes BLen, mem.GetPtr(0), StrPtr(A$)
        Set DecodeHEXtoMemBloc = mem
    ElseIf Len(A$) = 0 Then
        ok = False
        MyEr "Empty String to make buffer", "Αδειο αλφαριθμητικό για να φτιάξω διάρθρωση"
    Else
        MyEr "Problem with DecodeΗΕΧ", "Πρόβλημα αποκωδικοποίησης δεκαεξαδικού"
    End If
    
End Function

Public Sub CheckOrientation(A As cDIBSection, f As String)
    If LCase(ExtractType(f, (0))) = "jpg" Then
        Dim qw As New ExifRead
        qw.Load f
        Select Case qw.Tag(Orientation)
        Case 3
          RotateDib180 A
        Case 8
          RotateDib90 A
        Case 6
          RotateDib270 A
        End Select
    End If
End Sub

' OpenGL integration - I small step
Function PrepareLayer(thislayer As VB.PictureBox) As Boolean
Dim what As Object, parent1 As VB.PictureBox, parent2 As Form
If Not thislayer Is Nothing Then
    If nOpenGl.Ready Then
        nOpenGl.StopOpenGL
    End If
    If nOpenGl.Ready Then
        noUseOpenGL
        Exit Function
    End If
    Set what = thislayer.Parent
    If what Is Nothing Then
            noUseOpenGL
            Exit Function
    ElseIf TypeOf what Is VB.PictureBox Then
        thislayer.AutoRedraw = False  ' this is basic for opengl
        
        Set parent1 = what
        With parent1
        'thislayer.move .Left, .top, .width, .height
        .ScaleMode = vbPixels
        End With
        With nOpenGl
        If .Start(thislayer, parent1) Then
            .SetBackgroundColor 0!, 0!, 0!
            Call .SetBackgroundPictureUnder(parent1, thislayer.Left, thislayer.top)
        
            parent1.ScaleMode = vbTwips
        Else
            thislayer.AutoRedraw = True
            parent1.ScaleMode = vbTwips
            noUseOpenGL
            Exit Function
        End If
        End With
    Else
        thislayer.AutoRedraw = False  ' this is basic for opengl
        ' found if guiM2000 ........
        
        Set parent2 = what
        With parent2
       ' thislayer.move 0, 0 ', .width, .height
        '.ScaleMode = vbPixels
        End With
        With nOpenGl
        If .Start(thislayer, thislayer) Then
            .SetBackgroundColor 0!, 0!, 0!
            Call .SetBackgroundPictureUnder(parent2, thislayer.Left, thislayer.top)
            parent2.ScaleMode = vbTwips
        Else
            thislayer.AutoRedraw = True
            parent2.ScaleMode = vbTwips
            noUseOpenGL
            Exit Function
        End If
        End With
    End If
End If
PrepareLayer = True
End Function
